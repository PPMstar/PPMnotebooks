    sf = asct.readTable(shell_files_dir+shell_file,sldir='.')
    sfs[dump]['instance'] = sf

    sfs[dump]['t_m'] = t_m
    sfs[dump]['t_m_str'] = "{:.0f} min".format(t_m)
